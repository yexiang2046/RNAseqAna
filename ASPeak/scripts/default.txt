# No spaces between the commas
# If no lambda file is specified, then for each position i, the lambdas will be taken
# around some eps neighborhood, [i-eps,i+eps] by averaging the counts (in the wig  file) on this interval.
# The possible eps values are given in frameDiameters.
# For each i, some average values will be computed according to each given eps and the maximum
# will be  the lambda of i. The possible eps values are given separated by commas
frameRadii = 1000,5000,10000


# In the estimation of the parameters of the Negative Binomial (NB) distribution,
# the intervals are sorted by their rpkm value.
# For each interval estimationRadius many intervals above and estimationRadius intervals below the interval are taken.
# Among these interals the top 5% having the highest average read counts and 
# the bottom 5% having the lowest average read counts are removed to avoid outlier effects.
# This parameter is optional. When not given, 500 is taken to be its default value.
estimationRadius = 500
